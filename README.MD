## ğŸš€ Usage

```jsx
import React, { useState } from 'react';
import Clockify from 'react-clockify';

const App = () => {
  const [selectedTime, setSelectedTime] = useState('12:00');

  return (
    <div>
      <h3>Selected Time: {selectedTime}</h3>
      <Clockify 
        value={selectedTime} 
        onChange={(time) => setSelectedTime(time)} 
        format="12" // or "24"
      />
    </div>
  );
};

export default App;
```
